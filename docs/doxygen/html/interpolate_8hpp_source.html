40template <typename ExecSpace, int_t Dim>
41void Interpolate(Transfer<ExecSpace, Dim> &transfer) {
42 const TransferMethods method = transfer.method;
43
44 switch (method) {
45 case TransferMethods::NEAREST_NEAREST:
46 FiniteElements::FTNearest(transfer);
47 break;
48 case TransferMethods::INTERP_NEAREST:
49 FiniteElements::FTInterpNearest(transfer);
50 break;
51 case TransferMethods::INTERP_ZEROFILL:
52 FiniteElements::FTInterZero(transfer);
53 break;
54 case TransferMethods::INTERP_EXTRAP:
55 FiniteElements::FTInterExtrap(transfer);
56 break;
57 case TransferMethods::INTERP_CLAMP:
58 FiniteElements::FTInterClamp(transfer);
59 break;
60 default:
61 std::cerr << "Invalid transfer method" << std::endl;
62 assert(false);
63 break;
64 }
65 return;
66}
